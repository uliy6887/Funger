<script>
    // ========== 遊戲狀態 ==========
    let currentSize = 5;          // 3,5,6
    let currentGrid = [];          // 一維陣列
    let nextNumber = 1;
    let gameActive = true;
    let startTime = null;
    let timerInterval = null;
    let errorCount = 0;
    let bestTimes = {
        3: localStorage.getItem('bestTime_3') ? parseFloat(localStorage.getItem('bestTime_3')) : null,
        5: localStorage.getItem('bestTime_5') ? parseFloat(localStorage.getItem('bestTime_5')) : null,
        6: localStorage.getItem('bestTime_6') ? parseFloat(localStorage.getItem('bestTime_6')) : null
    };

    // DOM
    const gridContainer = document.getElementById('grid');
    const timerDisplay = document.getElementById('timerDisplay');
    const bestDisplay = document.getElementById('bestDisplay');
    const errorSpan = document.getElementById('errorCount');
    const resetBtn = document.getElementById('resetBtn');
    const shuffleBtn = document.getElementById('shuffleBtn');
    const messageDiv = document.getElementById('messageArea');
    const sizeBtns = document.querySelectorAll('.size-btn');

    // 更新最佳顯示
    function updateBestUI() {
        const best = bestTimes[currentSize];
        if (best !== null) bestDisplay.innerText = best.toFixed(2) + ' sec';
        else bestDisplay.innerText = '--';
    }

    // 儲存當前尺寸的最佳
    function saveBestTime(time) {
        const key = `bestTime_${currentSize}`;
        if (bestTimes[currentSize] === null || time < bestTimes[currentSize]) {
            bestTimes[currentSize] = time;
            localStorage.setItem(key, time);
            updateBestUI();
            return true;
        }
        return false;
    }

    // 停止計時
    function stopTimer() {
        if (timerInterval) {
            clearInterval(timerInterval);
            timerInterval = null;
        }
    }

    function startTimerIfNeeded() {
        if (startTime === null && nextNumber === 1 && gameActive) {
            startTime = Date.now();
            timerInterval = setInterval(() => {
                if (gameActive && startTime) {
                    const elapsed = (Date.now() - startTime) / 1000;
                    timerDisplay.innerText = elapsed.toFixed(2);
                }
            }, 100);
        }
    }

    // 產生亂序數字陣列 (1 ~ total)
    function generateShuffledArray(total) {
        const arr = Array.from({ length: total }, (_, i) => i + 1);
        for (let i = arr.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [arr[i], arr[j]] = [arr[j], arr[i]];
        }
        return arr;
    }

    // 渲染網格
    function renderGrid() {
        const size = currentSize;
        const total = size * size;
        gridContainer.style.gridTemplateColumns = `repeat(${size}, 1fr)`;
        gridContainer.innerHTML = '';
        for (let i = 0; i < currentGrid.length; i++) {
            const num = currentGrid[i];
            const cell = document.createElement('div');
            cell.className = 'cell';
            if (num < nextNumber) cell.classList.add('completed');
            cell.setAttribute('data-index', i);
            cell.innerText = num;
            cell.addEventListener('click', (e) => {
                e.stopPropagation();
                handleCellClick(i);
            });
            gridContainer.appendChild(cell);
        }
    }

    // 點擊邏輯
    function handleCellClick(index) {
        if (!gameActive) return;
        const clickedNum = currentGrid[index];
        if (clickedNum === nextNumber) {
            // 正確
            const cellDiv = document.querySelector(`.cell[data-index='${index}']`);
            if (cellDiv && !cellDiv.classList.contains('completed')) {
                cellDiv.classList.add('completed');
            }
            nextNumber++;
            startTimerIfNeeded();

            const total = currentSize * currentSize;
            if (nextNumber > total) {
                // 遊戲完成
                gameActive = false;
                stopTimer();
                const finalTime = (Date.now() - startTime) / 1000;
                timerDisplay.innerText = finalTime.toFixed(2);
                const isRecord = saveBestTime(finalTime);
                messageDiv.style.display = 'block';
                messageDiv.innerHTML = `✨ 完成！ 用時 ${finalTime.toFixed(2)} 秒 ${isRecord ? '🎉 新紀錄！ 🎉' : ''}`;
            }
        } else {
            // 錯誤
            errorCount++;
            errorSpan.innerText = errorCount;
            const wrongCell = document.querySelector(`.cell[data-index='${index}']`);
            if (wrongCell) {
                wrongCell.classList.add('error-flash');
                setTimeout(() => wrongCell.classList.remove('error-flash'), 300);
            }
        }
    }

    // 重置遊戲 (可選擇是否保留當前數字)
    function resetGame(keepNumbers = false) {
        stopTimer();
        messageDiv.style.display = 'none';
        if (!keepNumbers) {
            const total = currentSize * currentSize;
            currentGrid = generateShuffledArray(total);
        }
        nextNumber = 1;
        errorCount = 0;
        errorSpan.innerText = '0';
        gameActive = true;
        startTime = null;
        timerDisplay.innerText = '0.00';
        renderGrid();
    }

    // 洗牌 (強制新亂數，重置所有進度)
    function shuffleGame() {
        stopTimer();
        const total = currentSize * currentSize;
        currentGrid = generateShuffledArray(total);
        nextNumber = 1;
        errorCount = 0;
        errorSpan.innerText = '0';
        gameActive = true;
        startTime = null;
        timerDisplay.innerText = '0.00';
        messageDiv.style.display = 'none';
        renderGrid();
    }

    // 切換尺寸
    function setSize(newSize) {
        if (newSize === currentSize) return;
        currentSize = newSize;
        // 更新按鈕樣式
        sizeBtns.forEach(btn => {
            if (parseInt(btn.getAttribute('data-size')) === newSize) {
                btn.classList.add('active');
            } else {
                btn.classList.remove('active');
            }
        });
        // 重新產生該尺寸的亂數表
        const total = newSize * newSize;
        currentGrid = generateShuffledArray(total);
        nextNumber = 1;
        errorCount = 0;
        errorSpan.innerText = '0';
        gameActive = true;
        startTime = null;
        timerDisplay.innerText = '0.00';
        messageDiv.style.display = 'none';
        stopTimer();
        renderGrid();
        updateBestUI();
    }

    // LIFF 初始化
    async function initLIFF() {
        const LIFF_ID = '2010026891';   // 您提供的 ID
        try {
            await liff.init({ liffId: LIFF_ID });
            if (liff.isLoggedIn()) {
                const profile = await liff.getProfile();
                document.getElementById('userName').innerText = profile.displayName || 'LINE User';
                const avatarImg = document.getElementById('avatarImg');
                if (avatarImg && profile.pictureUrl) avatarImg.src = profile.pictureUrl;
            } else {
                document.getElementById('userName').innerText = '請登入 LINE';
                liff.login();
            }
        } catch (err) {
            console.error(err);
            document.getElementById('userName').innerText = '訪客模式';
        }
    }

    // 事件綁定
    function bindEvents() {
        resetBtn.addEventListener('click', () => resetGame(false));
        shuffleBtn.addEventListener('click', () => shuffleGame());
        sizeBtns.forEach(btn => {
            btn.addEventListener('click', (e) => {
                const size = parseInt(btn.getAttribute('data-size'));
                setSize(size);
            });
        });
    }

    // 啟動
    window.addEventListener('load', async () => {
        if (typeof liff !== 'undefined') await initLIFF();
        else document.getElementById('userName').innerText = 'SDK 錯誤';
        // 預設 5x5
        currentSize = 5;
        const total = currentSize * currentSize;
        currentGrid = generateShuffledArray(total);
        renderGrid();
        updateBestUI();
        bindEvents();
    });
</script>
